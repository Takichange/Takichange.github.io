```css
  /* 单行行文字超出变省略号 */
.ellipsis{
  overflow: hidden;
  text-overflow: ellipsis;
  white-space: nowrap;
}
   
  /* 多行文字超出变省略号 */
.ellipsis{
  word-break: break-all;
  overflow: hidden;          //  超出隐藏
  text-overflow: ellipsis;   //  超出显示省略号
  display: -webkit-box;
  /*! autoprefixer: off */
  -webkit-box-orient: vertical;
  /* autoprefixer: on */ // 有时orient不会生效所以需要加上这俩段注释
  -webkit-line-clamp: 5; // 超出省略行数
}

```
